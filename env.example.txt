# Task Tracker — Environment Configuration Reference

# Copy this file to .env and fill in the values for your environment.
# All variables without defaults MUST be set before starting the application.
#
# This file is used by:
#   - Docker Compose (PostgreSQL container and app container)
#   - FastAPI backend (via pydantic-settings / python-dotenv)

# ------------------------------------------------------------
# PostgreSQL Database (Docker Compose)
# Required when running via Docker Compose with the db service.
# The db service name resolves as the hostname within the Docker network.
# ------------------------------------------------------------

POSTGRES_DB=taskdb
POSTGRES_USER=taskdb

# Generate a secure password before deploying:
#   openssl rand -base64 32
POSTGRES_PASSWORD=your_secure_password_here


# ------------------------------------------------------------
# Backend: Database Connection URL
# ------------------------------------------------------------

# DATABASE_URL is constructed automatically by compose.yml and compose.traefik.yml
# using the POSTGRES_USER, POSTGRES_PASSWORD, and POSTGRES_DB values above.
# You do not need to set it here when running via Docker Compose.
#
# Local development only (no Docker — runs SQLite directly):
# DATABASE_URL=sqlite+aiosqlite:///./taskdb.db


# ------------------------------------------------------------
# Backend: Application Settings
# ------------------------------------------------------------

# ENVIRONMENT: Controls runtime behaviour.
#   development — enables hot-reload, verbose logging, relaxed error handling
#   production  — strict error handling, optimised for performance
ENVIRONMENT=production

# Development:
# ENVIRONMENT=development

# DEBUG: Enables FastAPI debug mode (true/false).
#   false — recommended for production (disables /docs, suppresses detailed errors)
#   true  — enables automatic API docs (/docs), detailed error pages, verbose output
DEBUG=false

# Development:
# DEBUG=true


# ------------------------------------------------------------
# Backend: CORS Allowed Origins
# ------------------------------------------------------------

# Comma-separated list of origins the browser is permitted to call the API from.
# Must be a JSON-style list of quoted strings.
#
# Note: when the frontend and API are served under the same domain via a reverse
# proxy (Traefik), the browser treats requests as same-origin and CORS is not
# enforced. This setting only matters when origins differ (e.g. local dev).

# Production — replace with your actual domain:
CORS_ORIGINS=["https://your-domain.com"]

# Local development (Vite dev server):
# CORS_ORIGINS=["http://localhost:5173","http://localhost:3000"]


# ------------------------------------------------------------
# Deployment: Application Domain (Traefik)
# ------------------------------------------------------------

# The public hostname Traefik uses to route HTTPS traffic to the app.
# Only required when deploying with compose.traefik.yml.
APP_DOMAIN=your-domain.com
